```python
class BuildReportDataChain(Runnable):
    def __init__(self, heading_tree: HeadingNode | None = None, target_lang: str = "zh") -> None:
        self.heading_tree = heading_tree
        self.target_lang = target_lang

    async def ainvoke(self, input: list[ArticleListItem], config=None) -> ReportData:
        items = input
        report_data = ReportData(...)
        report_data.add_articles(items, lambda a: a.tags[0] if a.tags else "unknown")
        report_data.build(self.heading_tree)
        return report_data
```
