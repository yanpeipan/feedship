    async def batch_summarize_articles(articles: list[dict]) -> list[dict]:
        """Summarize multiple articles in a single LLM call.

        Args:
            articles: List of dicts with 'title' and 'content' or 'description' keys.

        Returns:
            List of dicts with title, summary, quality_score (0-1), keywords.
        """
        if not articles:
            return []

        # Build article list string - truncate each to 500 words
        article_parts = []
        for i, art in enumerate(articles, 1):
            title = art.get("title", f"Article {i}")
            content = art.get("content") or art.get("description") or ""
            words = content.split()
            truncated = " ".join(words[:500]) if words else ""
            article_parts.append(f"[{i}] Title: {title}\nContent: {truncated}")

        article_list = "\n\n".join(article_parts)
        prompt = BATCH_SUMMARIZE_PROMPT.format(article_list=article_list)

        result = await llm_complete(prompt, max_tokens=2000, temperature=0.3)

        # Parse JSON - handle markdown code blocks
        import re
        json_str = re.sub(r"```json\n?|```\n?", "", result.strip())
        try:
            data = json.loads(json_str)
        except json.JSONDecodeError as e:
            logger.warning("batch_summarize_articles JSON parse failed: %s", e)
            raise

        # Normalize and validate
        results = []
        for item in data:
            score = float(item.get("quality_score", 50)) / 100
            score = min(max(score, 0.0), 1.0)
            results.append({
                "title": item.get("title", ""),
                "summary": item.get("summary", ""),
                "quality_score": score,
                "keywords": list(item.get("keywords", []))[:5],
            })

        return results
    ```

    all_processed: list[dict] = []
    semaphore = asyncio.Semaphore(5)

    # Collect articles needing summarization
    to_summarize: list[tuple] = []  # (article_dict, index)

    async def bounded_process(a: dict, idx: int) -> dict:
        async with semaphore:
            full = a
            summary = full.get("summary") or ""
            title = full.get("title", "")
            feed_weight = full.get("feed_weight", 0)
            feed_id = full.get("feed_id", "unknown")

            if not summary and auto_summarize and feed_weight >= 0.7:
                content = full.get("content") or full.get("description") or ""
                if content:
                    to_summarize.append((full, idx))

            return {
                "id": full["id"],
                "title": title,
                "link": full.get("link", ""),
                "summary": summary,
                "quality_score": full.get("quality_score"),
                "published_at": full.get("published_at"),
                "feed_id": feed_id,
                "original": full,
            }

    if articles:
        results = await asyncio.gather(
            *[bounded_process(a, i) for i, a in enumerate(articles)],
            return_exceptions=True,
        )
        for item in results:
            if isinstance(item, Exception):
                continue
            all_processed.append(item)

    # Batch summarize collected articles
    if to_summarize:
        batch_size = 3
        for i in range(0, len(to_summarize), batch_size):
            batch = [t[0] for t in to_summarize[i:i + batch_size]]
            try:
                batch_results = await batch_summarize_articles(batch)
                for orig, result in zip(batch, batch_results):
                    # Update all_processed entry
                    for proc in all_processed:
                        if proc["id"] == orig["id"]:
                            proc["summary"] = result.get("summary", "")
                            proc["quality_score"] = result.get("quality_score", 0.5)
                            break
                    # Collect for DB write
                    pending_writes_v2.append((
                        orig["id"],
                        result.get("summary", ""),
                        result.get("quality_score", 0.5),
                        [],
                        []
                    ))
            except Exception as e:
                logger.warning("Batch summarization failed, falling back to individual: %s", e)
                for art in batch:
                    try:
                        s, _, q, _ = await summarize_article_content(
                            art.get("content") or art.get("description") or "",
                            art.get("title", "")
                        )
                        for proc in all_processed:
                            if proc["id"] == art["id"]:
                                proc["summary"] = s
                                proc["quality_score"] = q
                                break
                        pending_writes_v2.append((art["id"], s, q, [], []))
                    except Exception as e2:
                        logger.error("Fallback summarization failed for %s: %s", art.get("id"), e2)
    ```
