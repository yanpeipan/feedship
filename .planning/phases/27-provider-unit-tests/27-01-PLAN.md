<interfaces>
<!-- RSSProvider Public Interface -->
RSSProvider methods:
- match(url: str) -> bool  # Uses httpx.head(), checks content-type header
- priority() -> int  # Returns 50
- crawl(url: str) -> List[Raw]  # Uses fetch_feed_content() which uses httpx.get()
- async crawl_async(url: str) -> List[Raw]  # Uses httpx.AsyncClient
- parse(raw: Raw) -> Article  # Converts feedparser entry to Article dict
- feed_meta(url: str) -> Feed  # Uses httpx.get() + feedparser.parse()

# RSSProvider.parse() output Article dict:
Article(
    title=raw.get("title"),
    link=raw.get("link"),
    guid=generate_article_id(raw),  # from src.utils
    pub_date=raw.get("published") or raw.get("updated"),
    description=raw.description or raw.summary,
    content=raw.content[0].value or raw.summary_detail.value,
)

<!-- GitHubReleaseProvider Public Interface -->
GitHubReleaseProvider methods:
- match(url: str) -> bool  # Checks for github.com in URL, supports HTTPS and SSH formats
- priority() -> int  # Returns 200
- crawl(url: str) -> List[Raw]  # Uses PyGithub get_latest_release()
- async crawl_async(url: str) -> List[Raw]  # Uses asyncio.to_thread(self.crawl, url)
- parse(raw: Raw) -> Article  # Converts release dict to Article dict
- feed_meta(url: str) -> Feed  # Uses PyGithub get_repo()

# GitHubReleaseProvider.parse() output Article dict:
Article(
    title=raw.get("tag_name") or raw.get("name"),
    link=raw.get("html_url"),
    guid=raw.get("tag_name"),
    pub_date=raw.get("published_at"),
    description=raw.get("body"),
    content=None,
)

<!-- ProviderRegistry Functions -->
- discover(url: str) -> List[ContentProvider]  # Returns providers where p.match(url) is True
- discover_or_default(url: str) -> List[ContentProvider]  # Falls back to RSSProvider if no match
- get_all_providers() -> List[ContentProvider]  # Returns PROVIDERS list sorted by priority desc

<!-- Raw Types -->
RSSProvider Raw = feedparser entry dict (has .get(), .get("title"), .link, .published, .updated, .description, .summary, .content, .summary_detail)
GitHubReleaseProvider Raw = dict with keys: tag_name, name, body, html_url, published_at
</interfaces>
